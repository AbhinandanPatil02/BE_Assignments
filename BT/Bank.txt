pragma solidity ^0.8.0;

contract Bank{

    uint256 balance=0;
    address public accOwner;

    constructor(){
        accOwner= msg.sender;
    }

    // Deposit
    function Deposit() public payable{
        require(accOwner==msg.sender,"You are not account owner");
        require(msg.value > 0,"Amount should be greater than zero");
        balance+=msg.value;
    }

    // Withdraw
    function Withdraw(uint256 amt) public payable {
        require(accOwner==msg.sender,"You are not account owner");
        require(amt>0,"Amount should be greater than zero");
        require(amt<=balance,"Insufficient balance");
        balance-=amt;
    }

    // Showbalance
    function ShowBalance() public view returns(uint256){
        require(accOwner==msg.sender,"You are not account owner");
        return balance;
    }

}
